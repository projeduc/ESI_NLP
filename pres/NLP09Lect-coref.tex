\begin{frame}
	\frametitle{\inserttitle}
	\framesubtitle{\insertshortsubtitle: Why still resolve coreferences?}
	
	\begin{itemize}
		\item \textbf{Long-range reasoning:} PLMs may fail to track entities across long documents.
		\item \textbf{Explainability:} Coreference chains provide interpretable links between mentions.
		\item \textbf{Improves downstream tasks:} 
		\begin{itemize}
			\item Summarization, QA, relation extraction
			\item Ensures pronouns are linked correctly
		\end{itemize}
		\item \textbf{Data efficiency:} Structured coreference features reduce the need for huge labeled datasets.
		\item \textbf{Domain adaptation:} PLMs may misinterpret pronouns or rare entities in specialized domains.
		\item \textbf{Explicit vs. implicit knowledge:} Guarantees consistent, repeatable entity linking across a document.
	\end{itemize}
	
\end{frame}

	\begin{block}{Reference [Merriam-Webster]}
		one referred to or consulted: such as
		\begin{itemize}
			\item A person to whom inquiries as to character or ability can be made
			\item A statement of the qualifications of a person seeking employment or appointment given by someone familiar with the person
			\item A source of information (such as a book or passage) to which a reader or consulter is referred
			\item A work (such as a dictionary or encyclopedia) containing useful facts or information

\subsection{Forms of reference}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}

	\begin{itemize}
		\item \optword{Pronouns}
		\begin{itemize}
			\item Personal: \expword{\underline{Karim} entered. \underline{\textbf{He}} started the class.}
			\item Possessive: \expword{\underline{Karim} has started \underline{\textbf{his}} course.}
			\item ...
		\end{itemize}

		\begin{itemize}
			\item \expword{I have a little \underline{cat}. \underline{\textbf{This animal}} is very small.}
		\end{itemize}
	
		\item \optword{Proper nouns} 
		\begin{itemize}

		\end{itemize}
	
		\item \optword{Zero anaphora} (omitted subject)
		\begin{itemize}
			\item \expword{\underline{Karim} presented and then \underline{\textbf{$ \phi $}} explained the course.}
			\item \expword{\begin{CJK}{UTF8}{min}\underline{カリムさん}はESIに行きます。\underline{\textbf{$ \phi $}} あそこに教えています。\end{CJK}}
		\end{itemize}
	\end{itemize}
	
\end{frame}

\subsection{Types of reference}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection}
	
	\begin{itemize}
		\item \optword{Anaphora}
		\begin{itemize}
			\item A reference to a previously mentioned expression; the referred expression is called the \keyword{antecedent}.
			\item E.g. \expword{\underline{The lesson} is very long. \underline{\textbf{It}} will take longer.}
		\end{itemize}
	
		\item \optword{Cataphora}
		\begin{itemize}
			\item A reference to a following expression; the referred expression is also called the \keyword{antecedent}.
			\item E.g. \expword{\underline{\textbf{It}} is very long, \underline{this course}.}
		\end{itemize}
	
		\item \optword{Plural antecedents}
		\begin{itemize}
			\item A referring expression that denotes multiple antecedents.
			\item E.g. \expword{\underline{The lesson} will be followed by \underline{an exercise}. \underline{\textbf{They}} are important to understand the subject.}
		\end{itemize}
	
		\optword{Coreferent noun phrases}
		\begin{itemize}
			\item Two or more noun phrases that refer to the same entity.
			\item E.g. \expword{\underline{\textbf{Some of my colleagues}} have been very supportive. \underline{\textbf{These people}} deserve my gratitude.}
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\subsection{Features of coreference relations}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Syntactic/morphological features}
	
	
	\begin{itemize}
		\item \optword{Number}: Singular, Dual, Plural.
		\begin{itemize}
			\item \expword{\underline{Lions} are chasing a \textit{deer}. \underline{\textbf{They}} are very fast.}
		\end{itemize}
	
		\item \optword{Person}: First, Second, Third.
		\begin{itemize}
			\item \expword{\underline{My brother}\textsubscript{1} and \underline{I}\textsubscript{2} repaired \underline{\textbf{his}}\textsubscript{1} bicycle and \underline{\textbf{mine}}\textsubscript{2}.}
		\end{itemize}
		
		\item \optword{Gender}: Masculine, Feminine, Neuter.
		\begin{itemize}
			\item \expword{When \textit{the girl} met her \underline{father}, \underline{\textbf{he}} was very happy.}
		\end{itemize}
		
		\item \optword{Binding theory constraints}: syntactic constraints governing the relation between pronouns/reflexives and their antecedents.
		
		\begin{itemize}
			\item \expword{Janet bought herself a bottle of fish sauce.} [herself = Janet]
			\item \expword{Janet bought her a bottle of fish sauce.} [her $\ne$ Janet]
		\end{itemize}
	
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Discourse and semantic preferences}
	
	\begin{itemize}
		
		\item \optword{Recency}: recently mentioned entities are more likely to be selected as antecedents.
		\begin{itemize}
			\item \expword{The doctor found an \textit{old map}. Ali found an even \underline{older one}. \underline{\textbf{It}} is about an island.}
		\end{itemize}
		
		\item \optword{Grammatical relations}: Subjects are more likely to be antecedents than objects.
		\begin{itemize}
			\item \expword{\underline{Karim} went to a restaurant with \textit{his friend}. \underline{\textbf{He}} asked for a couscous dish.}
		\end{itemize}
		
		\item \optword{Verb semantics (implicit causality)}: certain verbs bias the interpretation of pronouns toward specific participants (subject or object).
		\begin{itemize}
			\item \expword{\underline{Omar} telephoned \textit{Karim}. \underline{\textbf{He}} lost the laptop.}
			\item \expword{\textit{Omar} criticized \underline{Karim}. \underline{\textbf{He}} lost the laptop.}
		\end{itemize}
	\end{itemize}
	
\end{frame}

\subsection{Mention detection}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection}
	
	\begin{itemize}
		\item \optword{Mention extraction}
		\begin{itemize}
			\item Noun phrases, pronouns, named entities.
		\end{itemize}
	
		\item \optword{Mention classification}
		\begin{itemize}
			\item Using linguistic rules. 
			E.g. In the expression ``\expword{It is thought that ...}", the pronoun ``\expword{It}" is not a reference. 
			In this case, a list of cognitive verbs \expword{believe, think, etc.} can be used to filter this pronoun.
			\item Using machine learning: train models to detect valid mentions and filter out non-referential expressions.
		\end{itemize}
	
	\end{itemize}
	
\end{frame}


\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Rule-based \cite{2013-lee-al}}
	
	\begin{itemize}
		\item Remove a mention if there exists a longer mention with the same syntactic head.
		E.g., ``\expword{the software company}" vs. ``\expword{the software company situated in Algiers}". 
		
		\item Discard numerical entities such as percentages, monetary expressions, cardinals, and quantities.
		E.g., ``\expword{9\%}", ``\expword{100\$}", ``\expword{one hundred nine}".
		
		\item Eliminate mentions that are embedded in a larger quantified expression (quantifier + ``of"). 
		E.g., ``\expword{a total of 177 projects}", ``\expword{none of them}", ``\expword{millions of people}".
		
		\item Remove non-referential pronouns, such as pleonastic ``it''.
		E.g., ``\expword{It is thought that ...}", the pronoun ``it" is not a reference. 
		Linguistic rules, e.g., lists of cognitive verbs (\expword{believe, think, etc.}), can be used to detect these pronouns.
		
		\item Remove mentions that are adjectival forms of countries or abbreviations.
		E.g., ``\expword{American}", ``\expword{U.S.}", ``\expword{U.K.}".
		
	\end{itemize}
	
\end{frame}

\subsection{Mention linking}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Mention-Pair model}
	
	\begin{itemize}
		\item Detect whether two mentions are coreferent.
		\item Binary classification: coreferent vs. non-coreferent.
		\item \textbf{Problem}: transitivity conflicts can occur.  \\ 
		E.g. \expword{Ms Kennedy $ \leftarrow $ Kennedy, Kennedy $ \leftarrow $ He}
	\end{itemize}
	\begin{figure}
		\centering
		\hgraphpage[.8\textwidth]{mention-pair-exp.pdf}
		\caption{Example of a Mention-Pair coreference model \cite{2019-jurafsky-martin}}
	\end{figure}
	
\end{frame}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Mention-Ranking model}
	
	\begin{itemize}
		\item Among previous mentions, select the most likely antecedent.
		\[\hat{a} = \arg\max\limits_{a_i \in \{\epsilon, a_1, \ldots, a_{n-1}\}} P(a_i \mid a_n)\]
		\item $ \epsilon $ represents the case where the mention has no antecedent.
	\end{itemize}
	\begin{figure}
		\centering
		\hgraphpage{mention-rank-exp.pdf}
		\caption{Example of a Mention-Ranking model \cite{2019-jurafsky-martin}}
	\end{figure}
	
\end{frame}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection: Entity-mention model}
	
	\begin{itemize}
		\item Also called the cluster-ranking model.
		\item \textbf{Input}: clusters corresponding to two mentions.
		
		\item Determine whether the two clusters are compatible.
		
		\item Estimate whether one cluster is an antecedent of the other, as in the Mention-Rank model.
		
		\item If the two clusters correspond to the same entity, merge them.
		
		\item \textbf{Output}: final set of coreferent clusters.
		
	\end{itemize}
	
\end{frame}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection}
	
	\begin{itemize}
		\item \keyword{MUC}: \optword{Message Understanding Conference}
		\begin{itemize}
			\item Link-based metric. 
			Measures the number of shared coreference links between the reference and system output
		\end{itemize}
		
		\item \keyword{B\textsuperscript{3}}
		\begin{itemize}
			\item Mention-based metric. 
			Global precision and recall are calculated as the average of precision and recall over individual mentions
		\end{itemize}
		
		\item \keyword{CEAF}: \optword{Constrained Entity-Alignment F-Measure}
		\begin{itemize}
			\item Two versions: mention-based or entity-based, depending on the similarity function used.
			Each system mention or entity is aligned with exactly one reference mention or entity
		\end{itemize}
		
		\item \keyword{BLANC} 
		\begin{itemize}
			\item Link-based metric.
			Global precision and recall are computed based on both coreference and non-coreference links
		\end{itemize}
		
		\item \keyword{LEA}: \optword{Link-based Entity-Aware}
		\begin{itemize}
			\item Uses entity size as a weight to measure importance.
			Evaluates the correctness of links between mentions in resolved entities
		\end{itemize}
	\end{itemize}
\end{frame}

\subsection{Entity linking}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection}
	
	\begin{itemize}
		\item Associate a textual mention with its corresponding entry in a structured knowledge base.
		
		\item Provides additional contextual or real-world knowledge about entities.
		
		\item \keyword{Wikification}: links a mention to its corresponding Wikipedia page.
		
		\item Entity linking steps: 
		\begin{itemize}
			\item \optword{Mention detection}: identify all candidate knowledge-base entities corresponding to a mention using queries.
			\item \optword{Mention disambiguation}: select the most likely entity among candidates using machine learning.
		\end{itemize}
		
		\item Some APIs:
		\begin{itemize}
			\item FALCON: fast and accurate entity linking \url{https://labs.tib.eu/falcon/} 
			\item Dandelion API: entity extraction and linking \url{https://dandelion.eu}
		\end{itemize}
		
	\end{itemize}
	
\end{frame}

\subsection{Named entity recognition (NER)}

\begin{frame}
	\frametitle{\insertshortsubtitle: \insertsection}
	\framesubtitle{\insertsubsection}
	
	\begin{itemize}
		\item Locate and classify named entities in a text.
		
		\item \keyword{Named entity}: persons, locations, organizations, dates, quantities, and other proper nouns.
		
		\item It is a subtask of information extraction.
		
		\item Recognition techniques
		\begin{itemize}
			\item \optword{Rules}: 
			Use handcrafted rules and dictionaries of names to identify entities and determine their types.
			
			\item \optword{Features-based ML}:
			Use machine learning with features such as word embeddings, prefixes and suffixes, dictionary membership, and contextual features.
			
			\item \optword{Sequences labeling}: 
			Sequence labeling approaches, such as those using \keyword{IOB} annotation (see Chapter 5).
			
			
			\expword{\scriptsize
			$ \underbrace{Google}_{B-ORG} $ 
			$ \underbrace{LLC}_{I-ORG} $ 
			$ \underbrace{is}_{O} $ 
			$ \underbrace{founded}_{O} $ 
			$ \underbrace{in}_{O} $ 
			$ \underbrace{Silicon}_{B-LOC} $ 
			$ \underbrace{Valley}_{B-LOC} $ 
			$ \underbrace{by}_{O} $ 
			$ \underbrace{Larry}_{B-PER} $ 
			$ \underbrace{Page}_{I-PER} $ 
			$ \underbrace{and}_{O} $ 
			$ \underbrace{Sergey}_{B-PER} $ 
			$ \underbrace{Brin}_{I-PER} $
			}
		\end{itemize}
	\end{itemize}
	
\end{frame}
